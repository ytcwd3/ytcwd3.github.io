#### 1. 修改 script.js

在文件开头添加按需加载函数（参考 `script_optimized.js`）：

```javascript
// 数据缓存
const dataCache = {
  index: null,
  nintendo: null,
  sony: null,
  pc_android: null,
  other: null
};

// 分类映射
const categoryMapping = {
  '任天堂': 'nintendo',
  '索尼': 'sony',
  'PC及安卓': 'pc_android',
  '其他平台': 'other'
};

// 加载索引数据
async function loadIndexData() {
  if (dataCache.index) return dataCache.index;

  const res = await fetch('data/gameData_index.json');
  dataCache.index = await res.json();
  return dataCache.index;
}

// 按需加载分类数据
async function loadCategoryData(category) {
  const key = categoryMapping[category];
  if (dataCache[key]) return dataCache[key];

  const res = await fetch(`data/gameData_${key}.json`);
  dataCache[key] = await res.json();
  return dataCache[key];
}
```

#### 2. 修改 loadGameData 函数

```javascript
async function loadGameData() {
  try {
    // 首次只加载索引
    const indexData = await loadIndexData();

    console.log('✅ 索引加载成功', indexData.length);
  } catch (e) {

```

#### 3. 修改搜索函数

```javascript
async function search() {
  const keyword = gameSearch.value.trim().toLowerCase();

  if (selectedTag) {
    // 如果选择了分类，加载该分类数据
    const categoryData = await loadCategoryData(selectedTag.category);
    const results = categoryData.filter(g =>
      g.name.toLowerCase().includes(keyword)
    );
    displayResults(results);
  } else {
    // 搜索所有分类（需要加载所有数据）
    const allData = await Promise.all([
      loadCategoryData('任天堂'),
      loadCategoryData('索尼'),
      loadCategoryData('PC及安卓'),
      loadCategoryData('其他平台')
    ]);
    const merged = allData.flat();
    const results = merged.filter(g =>
      g.name.toLowerCase().includes(keyword)
    );
    displayResults(results);
  }
}
```

## 常见问题

### Q: 如果用户搜索时没选分类，会加载所有数据吗？
A: 是的，但这是按需的。只有在用户真正需要时才加载。

### Q: 索引文件还是很大（1MB），能再优化吗？
A: 可以！索引文件只需要包含游戏名称和分类，可以进一步精简。

### Q: 如何回退到原来的版本？
A: 保留原来的 `gameData.json`，修改 `loadGameData` 函数即可。
